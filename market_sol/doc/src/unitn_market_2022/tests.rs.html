#[cfg(test)]
mod tests {
    use crate::good::{good::Good, good_kind::GoodKind, consts::STARTING_QUANTITY};
    use crate::good::consts::{DEFAULT_EUR_USD_EXCHANGE_RATE, DEFAULT_EUR_YEN_EXCHANGE_RATE, DEFAULT_EUR_YUAN_EXCHANGE_RATE, DEFAULT_GOOD_KIND};
    use crate::good::good_error::{GoodKindError, GoodMergeError, GoodSplitError};

    #[test]
    fn default_exchange_rate_test() {
        assert!(DEFAULT_EUR_YEN_EXCHANGE_RATE > 0.0);
        assert!(DEFAULT_EUR_USD_EXCHANGE_RATE > 0.0);
        assert!(DEFAULT_EUR_YUAN_EXCHANGE_RATE > 0.0);
    }

    #[test]
    fn good_news_test() {
        //Since king and quantity fields are private, the getters are used to test the correctness of the functions
        //Test the new function with arbitrary arguments
        let new_good = Good::new(
            GoodKind::YUAN, 10.0
        );
        assert_eq!(new_good.get_qty(), 10.0);
        assert_eq!(new_good.get_kind(), GoodKind::YUAN);
        //Test that the DEFAULT_GOOD_KIND correctly initializes the Good
        let new_good_2 = Good::new(
            DEFAULT_GOOD_KIND, 12000.5
        );
        assert_eq!(new_good_2.get_qty(), 12000.5);
        assert_eq!(new_good_2.get_kind(), DEFAULT_GOOD_KIND);

        //Test the new function with a negative starting quantity
        let new_good_negative_qty = Good::new(
            GoodKind::EUR, -150.0
        );
        assert_eq!(new_good_negative_qty.get_qty(), 0.0);
        assert_eq!(new_good_negative_qty.get_kind(), GoodKind::EUR);

        //Test the new_default function
        let new_default = Good::new_default();
        assert_eq!(new_default.get_qty(), STARTING_QUANTITY);
        assert_eq!(new_default.get_kind(), DEFAULT_GOOD_KIND);

        //Test the new function with a GoodKind coming from GoodKind::from_str (actually a test for this function instead of the new)
        let good_from_str = Good::new(GoodKind::from_str("USD").unwrap(), 0.0);
        assert_eq!(good_from_str.get_qty(), 0.0);
        assert_eq!(good_from_str.get_kind(), GoodKind::USD);

        //Test GoodKind::from_str
        let random_good_king = GoodKind::from_str("wrongGoodKing");
        assert!(random_good_king.is_err());
        let correct_good_kind = GoodKind::from_str("YUAN");
        assert_eq!(correct_good_kind.unwrap(), GoodKind::YUAN);
    }

    /// test the merge functions, BY SMSE
    #[test]
    fn good_merge_test(){

        //create 3 goods
        let mut g1 = Good::new(GoodKind::EUR,100.0);
        let g2 = Good::new(GoodKind::EUR,200.0);
        let g3 = Good::new(GoodKind::EUR,300.0);

        //marge them
        let result1 = g1.merge(g2);
        let result2 = g1.merge(g3);

        //testing the returned values
        assert_eq!(result1.unwrap(),());
        assert_eq!(result2.unwrap(),());

        //testing the final good
        assert_eq!(g1.get_kind(),GoodKind::EUR);
        assert_eq!(g1.get_qty(),600.);

        //repeating the test with a different good kind and quantities

        //create 3 goods
        let mut g1 = Good::new(GoodKind::YUAN,10.0);
        let g2 = Good::new(GoodKind::YUAN,100.0);
        let g3 = Good::new(GoodKind::YUAN,1000.0);

        //marge them
        let result1 = g1.merge(g2);
        let result2 = g1.merge(g3);

        //testing the returned values
        assert_eq!(result1.unwrap(),());
        assert_eq!(result2.unwrap(),());

        //testing the final good
        assert_eq!(g1.get_kind(),GoodKind::YUAN);
        assert_eq!(g1.get_qty(),1110.);

        //creating 2 goods with different kinds
        let mut g1 = Good::new(GoodKind::YEN,100.0);
        let g2 = Good::new(GoodKind::USD,200.0);

        //create a copy of the 2 original goods
        let g1_copy = g1.clone();
        let g2_copy = g2.clone();

        //merge the goods
        let result = g1.merge(g2);

        match result {
            //give error fi the return is ok
            Result::Ok(_) => panic!("this should be an error, not an OK"),
            //if the result is an error
            Result::Err(error) => {
                match error {
                    GoodMergeError::DifferentKindsOfGood(returned_good) => {
                        //check if g2 is equal to the original one
                        assert_eq!(returned_good,g2_copy);
                    }
                }
            }
        }

        //check if g1 is equal to the original one
        assert_eq!(g1,g1_copy);

    }
    
    #[test]
    /// Test the split function - TSE WG
    fn good_split_test() {
        let mut g1 = Good::new(GoodKind::YEN, 1000.0);

        //test split function with a valid amount and verify if the returning good has the amount specified in the function
        assert_eq!(g1.split(300.0).unwrap(), Good::new(GoodKind::YEN, 300.0));

        //test if the amount of the good is correctly updated after the split
        assert_eq!(g1, Good::new(GoodKind::YEN, 700.0));

        //test split function with an amount that exceeds the amount of the good and the returning error NoEnoughQuantityToSplit
        assert_eq!(g1.split(10000.0).unwrap_err(), GoodSplitError::NotEnoughQuantityToSplit);

        //test split function with a negative amount and the returning error NonPositiveSplitQuantity
        assert_eq!(g1.split(-100.0).unwrap_err(), GoodSplitError::NonPositiveSplitQuantity);
    }
}
