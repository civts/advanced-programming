use std::cell::RefCell;
use std::rc::Rc;

use crate::event::notifiable::Notifiable;
use crate::good::good::Good;
use crate::good::good_kind::GoodKind;

use self::good_label::GoodLabel;

/// This trait is to be implemented by each team
pub trait Market : Notifiable {
    /// Initializes the market with random initial good amounts, whose total sum is
    /// [crate::good::consts::STARTING_CAPITAL] of the default good at the initial default rates
    fn new_random() -> Rc<RefCell<dyn Market>> where Self: Sized;

    /// Initializes the market for debugging purposes, with fixed initial good amounts
    /// of 100000 of the default good at the initial default rates
    fn new_with_quantities(eur: f32, yen: f32, usd: f32, yuan: f32) -> Rc<RefCell<dyn Market>> where Self: Sized;

    /// Initializes the market with initial good amounts as saved in the file at the provided path;
    /// if any i/o [Error] is encountered when reading the file, the function return a new instance created with new_random()
    fn new_file(path: &str) -> Rc<RefCell<dyn Market>> where Self: Sized;

    /// Documentation at: https://github.com/WG-AdvancedProgramming/market-protocol-specifications/blob/change_lock-buy_functions_logic/market-protocol-specifications.md#market-functionality
    fn get_name(&self) -> &'static str;

    fn get_budget(&self) -> f32;

    fn get_buy_price(&self, kind: GoodKind, quantity: f32) -> Result<f32, MarketGetterError>;

    fn get_sell_price(&self, kind: GoodKind, quantity: f32) -> Result<f32, MarketGetterError>;

    fn get_goods(&self) -> Vec<GoodLabel>;

    /// Documentation at: https://github.com/WG-AdvancedProgramming/market-protocol-specifications/blob/change_lock-buy_functions_logic/market-protocol-specifications.md#the-lock-function
    fn lock_buy(&mut self, kind_to_buy: GoodKind, quantity_to_buy: f32, bid: f32, trader_name: String) -> Result<String, LockBuyError>;

    // Documentation at: https://github.com/WG-AdvancedProgramming/market-protocol-specifications/blob/change_lock-buy_functions_logic/market-protocol-specifications.md#the-buy-function
    fn buy(&mut self, token: String, cash: &mut Good) -> Result<Good, BuyError>;

    /// Documentation at: https://github.com/WG-AdvancedProgramming/market-protocol-specifications/blob/change_lock-buy_functions_logic/market-protocol-specifications.md#the-lock-function-1
    fn lock_sell(&mut self, kind_to_sell: GoodKind, quantity_to_sell: f32, offer: f32, trader_name: String) -> Result<String, LockSellError>;

    /// Documentation at: https://github.com/WG-AdvancedProgramming/market-protocol-specifications/blob/change_lock-buy_functions_logic/market-protocol-specifications.md#the-sell-function
    fn sell(&mut self, token: String, good: &mut Good) -> Result<Good, SellError>;
}

/// Errors returned by [Market::lock_trader_buy_from_market], appearing in the same order they can be possibly returned.
#[derive(PartialEq, Clone, Debug)]
pub enum LockBuyError {
    /// `quantity_to_buy` is zero or less than zero.
    NonPositiveQuantityToBuy { negative_quantity_to_buy: f32 },
    /// `bid` is zero or less than zero.
    NonPositiveBid { negative_bid: f32 },
    /// Can be returned if the specific market implementation disallows goods with more than one lock and the trader is attempting to lock a good that is already locked.
    GoodAlreadyLocked { token: String },
    /// if the trader exceed the maximum total locks the market allow (for the deadlock prevention)
    MaxAllowedLocksReached,
    /// The market does not own the requested good kind in the requested quantity.
    InsufficientGoodQuantityAvailable { requested_good_kind: GoodKind, requested_good_quantity: f32, available_good_quantity: f32 },
    /// The trader's bid for the requested quantity of good kind is too low.
    BidTooLow { requested_good_kind: GoodKind, requested_good_quantity: f32, low_bid: f32, lowest_acceptable_bid: f32 },
}

/// Errors returned by [Market::trader_buy_from_market], appearing in the same order they can be possibly returned.
#[derive(PartialEq, Clone, Debug)]
pub enum BuyError {
    /// The market does not recognize the trader's token as valid.
    UnrecognizedToken { unrecognized_token: String },
    /// The market recognizes the trader's token as valid, but the lock on the good has expired.
    ExpiredToken { expired_token: String },
    /// The trader is trying to buy with a good whose kind is not default.
    GoodKindNotDefault { non_default_good_kind: GoodKind },
    /// The default good passed by the trader contains less than the pre-agreed quantity.
    InsufficientGoodQuantity { contained_quantity: f32, pre_agreed_quantity: f32 },
}

/// Errors returned by [Market::lock_trader_sell_to_market], appearing in the same order they can be possibly returned.
#[derive(PartialEq, Clone, Debug)]
pub enum LockSellError {
    /// `quantity_to_sell` is zero or less than zero.
    NonPositiveQuantityToSell { negative_quantity_to_sell: f32 },
    /// `offer` is zero or less than zero.
    NonPositiveOffer { negative_offer: f32 },
    /// Can be returned if the specific market implementation disallows goods with more than one lock and the trader is attempting to lock a good that is already locked.
    GoodAlreadyLocked { token: String },
    /// If the trader exceed the maximum total locks the market allow (for the deadlock prevention)
    MaxAllowedLocksReached,
    /// The trader's offer exceeds the quantity of default good kind the market owns.
    InsufficientDefaultGoodQuantityAvailable { offered_good_kind: GoodKind, offered_good_quantity: f32, available_good_quantity: f32 },
    /// The trader's offer for the quantity of good kind is too high.
    OfferTooHigh { offered_good_kind: GoodKind, offered_good_quantity: f32, high_offer: f32, highest_acceptable_offer: f32 },
}

/// Errors returned by [Market::trader_sell_to_market], appearing in the same order they can be possibly returned.
#[derive(PartialEq, Clone, Debug)]
pub enum SellError {
    /// The market does not recognize the trader's token as valid.
    UnrecognizedToken { unrecognized_token: String },
    /// The market recognizes the trader's token as valid, but the lock on the good has expired.
    ExpiredToken { expired_token: String },
    /// The good passed by the trader has not the agreed kind.
    WrongGoodKind { wrong_good_kind: GoodKind, pre_agreed_kind: GoodKind },
    /// The good passed by the trader contains less than the pre-agreed quantity.
    InsufficientGoodQuantity { contained_quantity: f32, pre_agreed_quantity: f32 },
}

#[derive(PartialEq, Clone, Debug)]
pub enum MarketGetterError{
    /// The asked quantity is negative
    NonPositiveQuantityAsked,
    /// The market dose not have the quantity the trader is asking for
    InsufficientGoodQuantityAvailable { requested_good_kind: GoodKind, requested_good_quantity: f32, available_good_quantity: f32 },
}
